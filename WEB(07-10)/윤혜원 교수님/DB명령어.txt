DB 명령어

DB 자료형
문자 varchar (바이트 지정)
날짜 date (지 알아서 DB 시간의로 해줌)
------------------------------------------------------------------------------

create database DB이름; // <- DB파일 만들기

use DB이름; // <- 나 지금 설정한 DB사용할 거야

show tables; // <- 지금 들어가 있는 DB 테이블 항목보여줘

create table 테이블 이름(
항목 자료형(바이트 지정),
항목 자료형(바이트 지정),
항목 자료형(바이트 지정),
항목 자료형(바이트 지정)
); // <- 선택한 DB에 테이블 항목 추가

drop table DB명 // <- DB 삭제
