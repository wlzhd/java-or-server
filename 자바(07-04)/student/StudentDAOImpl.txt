	public int insertStudent(StudentVO svo) {
		System.out.println("StudentDAOImpl insertStudent 함수 진입");
		// 0. 변수초기화 (DB접속정보, DB통로, 쿼리문, 실행문)
		Connection conn = null;
		PreparedStatement pstmt = null;
		String query = "";
		int result = 0;
		try {
			// 1. DB연동
			conn = DBConnection.getConnection();
			// 2. 쿼리작성
			query = "INSERT INTO STUDENTTABLE "
					+ "values(?,?,?,?,?,?,SYSDATE())";
			// 3. DB통로
			pstmt = conn.prepareStatement(query);
			pstmt.setString(1, svo.getName());
			pstmt.setString(2, svo.getBirth());
			pstmt.setString(3, svo.getSh());
			pstmt.setString(4, svo.getCre());
			pstmt.setString(5, svo.getAddr());
			pstmt.setString(6, svo.getPhone());
			// pstmt.setString(7, svo.getRedate());
			System.out.println("pstmt >> " + pstmt);
			// 4. 쿼리실행 & 결과
			result = pstmt.executeUpdate(); // <- query 이거 넣으면 큰일난다
		}catch(SQLException e){
			System.out.println("DB오류 >> " + e);
		}
		return result;
	}
